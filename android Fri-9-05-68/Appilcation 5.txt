package com.example.myapplication

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
import android.widget.CheckBox
import android.widget.TextView

class MainActivity : AppCompatActivity() {
    lateinit var button: Button
    lateinit var chkBento : CheckBox
    lateinit var chkTeriyaki : CheckBox
    lateinit var chkTempura : CheckBox
    lateinit var chkRamen : CheckBox
    lateinit var chkSushi : CheckBox
    lateinit var textView: TextView
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        chkTeriyaki = findViewById(R.id.chkTeriyaki)
        chkBento = findViewById(R.id.chkBento)
        chkTempura = findViewById(R.id.chkTempura)
        chkRamen = findViewById(R.id.chkRamen)
        chkSushi = findViewById(R.id.chkSushi)
        button = findViewById(R.id.button1)
        textView = findViewById(R.id.textview1)
        button.setOnClickListener {
            val answer = StringBuffer()
            answer.append("\nBento check: ").append(chkBento.isChecked())
            answer.append("\nTeriyaki check: ").append(chkTeriyaki.isChecked())
            answer.append("\nTempura check: ").append(chkTempura.isChecked())
            answer.append("\nRamen check: ").append(chkRamen.isChecked())
            answer.append("\nSushi check: ").append(chkSushi.isChecked())

            textView.setText(answer)
        }
    }
}
